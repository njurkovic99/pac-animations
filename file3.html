#include <iostream> #include <fstream> #include <cstring> using namespace std; struct Record { int id; char name[20]; double score; }; int main() { // Create initial file with 3 records fstream file("data.bin", ios::out | ios::binary); Record r1 = {101, "Alice", 95.5}; Record r2 = {102, "Bob", 87.3}; Record r3 = {103, "Carol", 92.1}; file.write((char*)&r1, sizeof(Record)); file.write((char*)&r2, sizeof(Record)); file.write((char*)&r3, sizeof(Record)); file.close(); cout << "Created file with 3 records" << endl; // Open for both reading and writing file.open("data.bin", ios::in | ios::out | ios::binary); // Read second record (Bob) Record temp; file.seekg(1 * sizeof(Record), ios::beg); cout << "Position after seekg: " << file.tellg() << endl; file.read((char*)&temp, sizeof(Record)); cout << "Read: " << temp.name << " - " << temp.score << endl; // Update Bob's score to 91.8 temp.score = 91.8; // Seek back to Bob's position file.seekp(1 * sizeof(Record), ios::beg); cout << "Position after seekp: " << file.tellp() << endl; // Write updated record file.write((char*)&temp, sizeof(Record)); cout << "Updated Bob's score to 91.8" << endl; file.close(); // Verify with ios::ate (open at end) file.open("data.bin", ios::in | ios::binary | ios::ate); cout << "File size: " << file.tellg() << " bytes" << endl; // Read all records from beginning file.seekg(0, ios::beg); cout << "All records after update:" << endl; for(int i = 0; i < 3; i++) { file.read((char*)&temp, sizeof(Record)); cout << temp.id << ": " << temp.name; cout << " - " << temp.score << endl; } file.close(); return 0; }
