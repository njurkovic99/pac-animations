Correct and Incorrect Structure Member Access
C++ Code
#include <iostream> #include <string> using namespace std; struct Student { string name; int age; double gpa; }; int main() { // Create structure variable Student student1; student1.name = "Alice"; student1.age = 20; student1.gpa = 3.8; // Create pointer to structure Student* ptr = &student1; // CORRECT: Access via structure variable cout << student1.name << endl; // CORRECT: Access via pointer (arrow) cout << ptr->name << endl; // CORRECT: Access via pointer (dereference) cout << (*ptr).name << endl; // ERROR: Wrong operator on pointer // cout << ptr.name << endl; // ERROR: Wrong operator on variable // cout << student1->name << endl; return 0; }
