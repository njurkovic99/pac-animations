Enumerations and Unions
C++ Code
#include <iostream> #include <string> using namespace std; // Enumeration example enum Color { RED = 0, // 0 GREEN = 1, // 1 BLUE = 2 // 2 }; // Union example - shares memory union Data { int i; // 4 bytes float f; // 4 bytes char c[4]; // 4 bytes }; int main() { // Working with enums Color myColor = RED; cout << "Color value: " << myColor << endl; myColor = BLUE; cout << "New color: " << myColor << endl; // Working with unions Data data; cout << "Union size: " << sizeof(data) << endl; data.i = 1065353216; // Same bits as 1.0f cout << "As int: " << data.i << endl; cout << "As float: " << data.f << endl; data.f = 3.14f; cout << "Set float to 3.14" << endl; cout << "As int: " << data.i << endl; cout << "As float: " << data.f << endl; return 0; }
